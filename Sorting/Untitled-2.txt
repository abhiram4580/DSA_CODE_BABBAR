#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main(){
   int arr[]={3,2,2,1};
   int N=4;
   int i=N;
   int j=N;
   int l,n;
   while(i<N && j<N){
        if(arr[i]%2!=0){
            l=arr[i];
           }
        if(arr[j]%2!=1){
            n=arr[j];
        }
           swap(l,n);
           i=i+2;
           j=j+2;
 }

  for(int i=0;i<N;i++){
    cout<<arr[i]<<endl;
  }

  return 0;


}
